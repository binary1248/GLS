802  void introspect() {
803  // Attributes
804  {
805  m_attribute_map.clear();
806 
807  auto active_attributes = GLint();
808  check_gl_error( glGetProgramiv( name(), GL_ACTIVE_ATTRIBUTES, &active_attributes ) );
809 
810  auto active_attribute_max_length = GLint();
811  check_gl_error( glGetProgramiv( name(), GL_ACTIVE_ATTRIBUTE_MAX_LENGTH, &active_attribute_max_length ) );
812 
813  auto attribute_name = std::vector<GLchar>( static_cast<std::size_t>( active_attribute_max_length ), 0 );
814 
815  for( GLuint index = 0; index < static_cast<GLuint>( active_attributes ); ++index ) {
816  auto attribute_type = GLenum();
817  auto attribute_size = GLint();
818 
819  check_gl_error( glGetActiveAttrib( name(), index, active_attribute_max_length, nullptr, &attribute_size, &attribute_type, attribute_name.data() ) );
820 
821  auto attribute_location = GLint();
822 
823  attribute_location = check_gl_error( glGetAttribLocation( name(), attribute_name.data() ) );
824 
825  assert( attribute_location >= 0 );
826 
827  m_attribute_map.emplace( std::string( attribute_name.data() ), std::make_tuple( attribute_location, attribute_type, attribute_size ) );
828  }
829  }
830 
831  // Uniforms
832  {
833  m_uniform_map.clear();
834 
835  auto active_uniforms = GLint();
836  check_gl_error( glGetProgramiv( name(), GL_ACTIVE_UNIFORMS, &active_uniforms ) );
837 
838  auto active_uniform_max_length = GLint();
839  check_gl_error( glGetProgramiv( name(), GL_ACTIVE_UNIFORM_MAX_LENGTH, &active_uniform_max_length ) );
840 
841  auto uniform_name = std::vector<GLchar>( static_cast<std::size_t>( active_uniform_max_length ), 0 );
842 
843  for( GLuint index = 0; index < static_cast<GLuint>( active_uniforms ); ++index ) {
844  auto uniform_type = GLint();
845  auto uniform_size = GLint();
846 
847  check_gl_error( glGetActiveUniformName( name(), index, active_uniform_max_length, nullptr, uniform_name.data() ) );
848  check_gl_error( glGetActiveUniformsiv( name(), 1, &index, GL_UNIFORM_TYPE, &uniform_type ) );
849  check_gl_error( glGetActiveUniformsiv( name(), 1, &index, GL_UNIFORM_SIZE, &uniform_size ) );
850 
851  auto uniform_location = GLint();
852 
853  uniform_location = check_gl_error( glGetUniformLocation( name(), uniform_name.data() ) );
854 
855  assert( uniform_location >= 0 );
856 
857  m_uniform_map.emplace( std::string( uniform_name.data() ), std::make_tuple( uniform_location, static_cast<GLenum>( uniform_type ), uniform_size ) );
858  }
859  }
860 
861  // Uniform blocks
862  {
863  m_uniform_block_map.clear();
864 
865  auto active_uniform_blocks = GLint();
866  check_gl_error( glGetProgramiv( name(), GL_ACTIVE_UNIFORM_BLOCKS, &active_uniform_blocks ) );
867 
868  auto active_uniform_block_max_name_length = GLint();
869  check_gl_error( glGetProgramiv( name(), GL_ACTIVE_UNIFORM_BLOCK_MAX_NAME_LENGTH, &active_uniform_block_max_name_length ) );
870 
871  auto uniform_block_name = std::vector<GLchar>( static_cast<std::size_t>( active_uniform_block_max_name_length ), 0 );
872 
873 #if !defined( NDEBUG )
874  if( active_uniform_blocks > m_max_uniform_buffer_bindings ) {
875  GLS_ERROR_STREAM << "Error: Couldn't bind all uniform blocks." << std::endl;
876  GLS_ERROR_STREAM << "Active blocks: " << active_uniform_blocks << " Max bindings: " << m_max_uniform_buffer_bindings << std::endl;
877 
878  active_uniform_blocks = m_max_uniform_buffer_bindings;
879  }
880 #endif
881 
882  for( GLuint index = 0; index < static_cast<GLuint>( active_uniform_blocks ); ++index ) {
883  auto uniform_block_size = GLint();
884 
885  check_gl_error( glGetActiveUniformBlockName( name(), index, active_uniform_block_max_name_length, nullptr, uniform_block_name.data() ) );
886  check_gl_error( glGetActiveUniformBlockiv( name(), index, GL_UNIFORM_BLOCK_DATA_SIZE, &uniform_block_size ) );
887 
888  m_uniform_block_map.emplace( std::string( uniform_block_name.data() ), std::make_tuple( index, uniform_block_size ) );
889  }
890  }
891  }
892 
893  object<priv::create_program, priv::delete_program> m_object;
894  std::unordered_map<std::string, std::tuple<GLint, GLenum, GLint>> m_attribute_map;
895  std::unordered_map<std::string, std::tuple<GLint, GLenum, GLint>> m_uniform_map;
896  std::unordered_map<std::string, std::tuple<GLuint, GLint>> m_uniform_block_map;
897 
898  GLint m_max_uniform_buffer_bindings = 0;
899 };
